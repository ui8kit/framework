import { ComponentExample, CodeBlock, Callout } from '@ui8kit/mdx-react'
import { Button } from '@ui8kit/core'

# Basic MDX Example

This is a basic example of using @ui8kit/mdx-react for component documentation.

## Button Component

<ComponentExample
  title="Primary Button"
  component={Button}
  variants={['primary']}
  code={`<Button variant="primary">Click me</Button>`}
/>

## Code Example

<CodeBlock language="tsx" showLineNumbers>
{`import { Button } from '@ui8kit/core'

function MyComponent() {
  return (
    <Button variant="primary" onClick={() => alert('Clicked!')}>
      Click me
    </Button>
  )
}

export default MyComponent`}
</CodeBlock>

<Callout type="info">
  This is an informational callout. You can use different types: info, warning, error, success.
</Callout>
